# Assignment operators =
# =	: x = 5 : x = 5
# += : x += 5 : x = x + 5
# -= : x -= 5 : x = x - 5
# *= : x *= 5 : x = x * 5
# /= : x /= 5 : x = x / 5
# %= : x %= 5 : x = x % 5
# //= : x //= 5	: x = x // 5
# **= : x **= 5	: x = x ** 5
# &= : x &= 5 : x = x & 5
# |= : x |= 5 : x = x | 5
# ^= : x ^= 5 : x = x ^ 5
# >>= : x >>= 5	: x = x >> 5
# <<= : x <<= 5	: x = x << 5
# --------------------------------------------------
# Identity operators = Two variables that are equal does not imply that they are identical.
# Identity operators : Two variables that are equal does not imply that they are identical... This Refers to Address of Object.
# is = True if the operands are identical (refer to the same object) = x is True
# is not = True if the operands are not identical (do not refer to the same object)	= x is not True
# --------------------------------------------------
# Membership operators = used to test whether a value or variable is found in a sequence (string, list, tuple, set and dictionary).
# In a dictionary we can only test for presence of key, not the value.
# Membership operators : This Refers to Contents of Object.
# in = True if value/variable is found in the sequence = 5 in x
# not in = True if value/variable is not found in the sequence = 5 not in x
# --------------------------------------------------
